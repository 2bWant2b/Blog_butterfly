# 用户界面

# Widgets (基础组件)

main.dart 文件中是 Flutter 的入口方法，一个简单的例子如下：

import 'package:flutter/material.dart';

void main() {
  runApp(
    Center(
      child: Text(
        'Hello, world!',
        textDirection: TextDirection.ltr,
      ),
    ),
  );
}

这些组件一看就懂了，其中 textDirection: TextDirection.ltr 是文字显示从左向右，当然这是一种不合理的写法。可以写成这样：

void main() {
  runApp(MyApp());
}

//自定义一个组件
class MyApp extends StatelessWidget{
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text(
        'Hello, world!',
        textDirection: TextDirection.ltr,
        style: TextStyle(
          fontSize: 40,
          color: Colors.deepOrange,
      ),
    );
  }
}

一般实例化类 (组件都是类) 时省去 new 关键字。可以通过一些参数指定 Text 的属性 StatelessWidget 是无状态组件类.
下面是一种比较标准的写法：

void main() {
  runApp(MyApp());
}

//自定义一个组件
class MyApp extends StatelessWidget{
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
            title: Text('Flutter Demo')
        ),
        body: HomeContent(),
      ),
    );
  }
}

class HomeContent extends StatelessWidget{
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Text(
          'Hello flutter',
          textDirection: TextDirection.ltr,
          style: TextStyle(
              fontSize: 40,
              color: Color.fromRGBO(215, 546, 213, 0.8)
          )
      ),
    );
  }
}

以上是基础结构.

动态列表（举例非实战）

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Flutter Demo')),
        body: HomeContent(),
      ),
    );
  }
}

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ListView(
      children: _getData(),
    );
  }
//自定义一个返回数组的方法，传入上面的ListView中
  List<Widget> _getData(){
      List<Widget> list=new List();
      for(var i=0;i<14;i++){
        list.add(
          ListTile(title: Text("动态列表$i")),
        );
      }
      return list;
  }
}

然后我们可以来看看获取其他的数据来创建一个列表。
比如我们有一个资源数据如下

List listData=[
  {
    "name":"uzi",
    "age":25,
    "height":180
  },
  {
    "name":"faker",
    "age":27,
    "height":179
  },
  {
    "name":"clearlove",

  }
];

这里的列表里是 map 类型的数据，不知道可不可以看成 json 呢？

然后我们要先导入该文件

import 'res/test.dart';

然后写如下代码

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ListView(
      children: _getData(),
    );
  }
//自定义一个返回数组的方法，传入上面的ListView中
  List<Widget> _getData(){
      var tempList=listData.map((value){
        return ListTile(
          title: Text(value["name"]),
          subtitle: Text(value["age"].toString()),
        );
      });
      return tempList.toList();
  }
}

上面这玩意还可以用 ListView.builder 来写

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: listData.length,
      itemBuilder: _getData,
    );
  }
//自定义一个返回数组的方法，传入上面的ListView中
  Widget _getData(context,index){
      return ListTile(
        title: Text(listData[index]["name"]),
        subtitle: Text(listData[index]["age"].toString()),
      );
  }
}

# GridView

和 ListView 大同小异
基础用法：

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return GridView.count(
        crossAxisSpacing: 20, //水平Widght之间间距
        mainAxisSpacing: 20,  //垂直Widght之间间距
        padding: EdgeInsets.all(10),
        childAspectRatio: 0.6,  //宽高比
        crossAxisCount: 3,  //一行的数量
        children: _getListData(),
    );
  }
//自定义一个返回数组的方法，传入上面的GridView中
  List<Widget> _getListData(){
    List<Widget> list= new List();

    for (var i=0;i<20;i++){
      list.add(Container(
        alignment: Alignment.center,
        child: Text(
          "this is $i data",
          style: TextStyle(color: Colors.white,fontSize: 20),
        ),
        color: Colors.blue,
      ));
    }
    return list;
  }
}

同样的，写入外来数据：

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return GridView.count(
        crossAxisSpacing: 20, //水平Widght之间间距
        mainAxisSpacing: 20,  //垂直Widght之间间距
        padding: EdgeInsets.all(10),
        childAspectRatio: 0.6,  //宽高比
        crossAxisCount: 3,  //一行的数量
        children: _getListData(),
    );
  }
//自定义一个返回数组的方法，传入上面的GridView中
  List<Widget> _getListData(){

    var tempList=listData.map((value){
      return Container(
        child: Column(
          children: [
            Text(value["name"]),
            Text(value["height"].toString()),
          ],
        ),
      );
    });
    return tempList.toList();
  }
}

这里提前注释一下，map 方法的 value 就是列表里的每一个子元素，不是标号

然后呢就是 GridView.builder 啦

class HomeContent extends StatelessWidget {

      ),
    );

//自定义一个返回数组的方法，传入上面的GridView中

# Row（水平布局）

先看看源码：

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Row(
      mainAxisAlignment: MainAxisAlignment.center,
      children: [
        IconContainer(Icons.search,bgcolor: Colors.black,iconcolor: Colors.white,size: 35,),
        IconContainer(Icons.home,bgcolor: Colors.red,iconcolor: Colors.blue,size: 35,),
        IconContainer(Icons.settings,bgcolor: Colors.green,iconcolor: Colors.yellow,size: 35,),
      ],
    );
  }
}

class IconContainer extends StatelessWidget {
  double size=35.0;
  Color bgcolor=Colors.red;
  Color iconcolor;
  IconData icon;
  /*定义一个构造函数，用来设定该组件的属性*/
  IconContainer(this.icon,{this.bgcolor,this.size,this.iconcolor});

  @override
  Widget build(BuildContext context) {
     return Container(
       height: 100.0,
       width: 100.0,
       color: this.bgcolor,
       child: Center(
         child: Icon(this.icon,size:this.size,color: this.iconcolor,),
       ),
     );
  }

}

上面自定义了一个组件，就是一个 Container 里放了一个 icon ，并且写了构造函数方便设置属性。直接看看运行结果图吧。

# Expanded

用于给布局里的子组件设置权重，这个和原生开发的 weight 好像是差不多的，我觉得这个是套在比较外层的。直接看源码吧。

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Row(
      children: [
        Expanded(
          flex: 1,
          child: IconContainer(Icons.search,bgcolor: Colors.blue,iconcolor: Colors.white,size: 35,),
        ),
        Expanded(
          flex: 2,
            child: IconContainer(Icons.home,bgcolor: Colors.red,iconcolor: Colors.blue,size: 35,),
        ),
        Expanded(
          flex: 1,
          child: IconContainer(Icons.settings,bgcolor: Colors.amber,iconcolor: Colors.cyanAccent,size: 35,),
        ),
      ],
    );
  }
}

class IconContainer extends StatelessWidget {
  double size=35.0;
  Color bgcolor=Colors.red;
  Color iconcolor;
  IconData icon;
  /*定义一个构造函数，用来设定该组件的属性*/
  IconContainer(this.icon,{this.bgcolor,this.size,this.iconcolor});

  @override
  Widget build(BuildContext context) {
     return Container(
       height: 100.0,
       width: 100.0,
       color: this.bgcolor,
       child: Center(
         child: Icon(this.icon,size:this.size,color: this.iconcolor,),
       ),
     );
  }

}

SizedBox(height: 10) 或者 SizedBox(width:10) 用于指定布局中的子组件间的间隔，写在两个子组件中间即可，这里就先不写演示了

# Stack（层叠组件）

这实现了一种堆叠效果，处于同一位置的组件是可以被覆盖掉的。 Stack 经常搭配 Align (对齐) 一起食用效果更佳，实现定位布局。

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Center(
      child: Container(
        height: 400,
        width: 300,
        color: Colors.redAccent,
        child: Stack(
          children: [
            Align(
              alignment: Alignment(0, 0),
              child: Icon(Icons.autorenew,size: 40,color: Colors.blue,),
            ),
            Align(
              alignment: Alignment(-0.8, 0.7),
              child: Icon(Icons.home,size: 60,color: Colors.cyanAccent),
            ),
            Align(
              alignment: Alignment(-0.4, 0.3),
              child: Icon(Icons.search,size: 50,color: Colors.amber),
            ),
          ],
        ),
      ),
    );
  }
}

# AspectRatio

根据设置调整子组件 child 的宽高比，它是设置上级组件中的子组件宽高比。

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      width: 300,
      child: AspectRatio(
        aspectRatio: 1.5,
        child: Container(
          color: Colors.cyanAccent,
        ),
      ),
    );
  }
}

# Card 组件

制作一张卡片，经常外层套 ListView ，中间写 Card ，里面写 Column ，当然这不是固定的。

class HomeContent extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ListView(
      children: [
        Card(
          margin: EdgeInsets.all(15),
          child: Column(
            children: [
              AspectRatio(
                aspectRatio: 16/9,
                child: Image.network("https://www.itying.com/images/flutter/2.png",fit: BoxFit.cover,),
              ),
              ListTile(
                leading: CircleAvatar(
                    backgroundImage: NetworkImage("https://tse4-mm.cn.bing.net/th/id/OIP.BnQS9QMPBxhViMCXiFqB0QHaEZ?w=273&h=180&c=7&o=5&dpr=1.25&pid=1.7",)
                ),
                title: Text("Uzi",style: TextStyle(fontSize: 28),),
                subtitle: Text("ADC"),
              ),
            ],
          ),
        ),
        Card(
          margin: EdgeInsets.all(15),
          child: Column(
            children: [
              AspectRatio(
                aspectRatio: 16/9,
                child: Image.network("https://www.itying.com/images/flutter/3.png",fit: BoxFit.cover,),
              ),
              ListTile(
                leading: ClipOval(
                    child: Image.network("https://ss1.bdstatic.com/70cFvXSh_Q1YnxGkpoWK1HF6hhy/it/u=2160807766,800949609&fm=26&gp=0.jpg",fit: BoxFit.cover,height: 60,width: 55,)
                ),
                title: Text("Clearlove",style: TextStyle(fontSize: 28),),
                subtitle: Text("Jungle"),
              ),
            ],
          ),
        ),
      ],
    );
  }
}

这里设置 LIstTile 的 leading 时，用了两种方法。一是用 CircleAvatar 直接设置成头像，二是用 ClipOval 裁剪图片来设置成头像。
